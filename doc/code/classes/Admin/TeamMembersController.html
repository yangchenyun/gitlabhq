def update
  @admin_team_member = UsersProject.find(params[:id])

  if @admin_team_member.update_attributes(params[:team_member])
    redirect_to [:admin, @admin_team_member.project],  notice: 'Project Access was successfully updated.'
  else
    render action: "edit"
  end
end
