def create
  @hook = @project.hooks.new(params[:hook])
  @hook.save

  if @hook.valid?
    redirect_to project_hooks_path(@project)
  else
    @hooks = @project.hooks.all
    render :index
  end
end

def destroy
  @hook = @project.hooks.find(params[:id])

  redirect_to project_hooks_path(@project)
end
